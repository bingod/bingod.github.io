public class Solution {
    public int firstMissingPositive(int[] A) {
        for (int i = 0; i < A.length; i++) {
            if (A[i] == 0) A[i] = -1;
        }
        for (int i = 0; i < A.length; i++) {
            int p = A[i];
            while (p > 0 && p <= A.length) {
                int t = A[p - 1];
                A[p -1] = 0;
                p = t;
            }
        }
        for (int i = 0; i < A.length;i++) {
            if (A[i] != 0) return i + 1;
        }
        return A.length + 1;
    }
}
